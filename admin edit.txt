<?php
session_start();
include "../includes/db.php";

if (!isset($_SESSION['user_id'])) { 
    header("Location: index.php"); 
    exit; 
}

$id = intval($_GET['id']);
$paket = mysqli_query($conn, "SELECT * FROM packages WHERE id=$id");
$data = mysqli_fetch_assoc($paket);

if (!$data) {
    echo "Paket tidak ditemukan!";
    exit;
}

// Update Paket
if (isset($_POST['update'])) {
    $name   = mysqli_real_escape_string($conn, $_POST['name']);
    $region = mysqli_real_escape_string($conn, $_POST['region']);
    $days   = intval($_POST['days']);
    $price  = floatval($_POST['price']);

    $image = $data['image'];
    if (!empty($_FILES['image']['name'])) {
        // Hapus gambar lama
        if ($image && file_exists("../uploads/".$image)) {
            unlink("../uploads/".$image);
        }
        $targetDir = "../uploads/";
        $image = time() . "_" . basename($_FILES["image"]["name"]);
        $targetFile = $targetDir . $image;
        move_uploaded_file($_FILES["image"]["tmp_name"], $targetFile);
    }

    mysqli_query($conn, "UPDATE packages SET name='$name', region='$region', days='$days', price='$price', image='$image' WHERE id=$id");
    header("Location: packages.php");
    exit;
}
?>
<!DOCTYPE html>
<html lang="id">
<head>
  <meta charset="UTF-8">
  <title>Edit Paket</title>
  <link href="https://cdn.jsdelivr.net/npm/tailwindcss@2.2.19/dist/tailwind.min.css" rel="stylesheet">
</head>
<body class="bg-gray-100">
  <div class="max-w-xl mx-auto p-6">
    <h2 class="text-2xl font-bold mb-4">Edit Paket Wisata</h2>
    <form method="POST" enctype="multipart/form-data" class="bg-white p-4 rounded shadow grid grid-cols-2 gap-3">
      <input type="text" name="name" value="<?php echo htmlspecialchars($data['name']); ?>" required class="border p-2 rounded col-span-2">
      <input type="text" name="region" value="<?php echo htmlspecialchars($data['region']); ?>" required class="border p-2 rounded col-span-1">
      <input type="number" name="days" value="<?php echo $data['days']; ?>" required class="border p-2 rounded col-span-1">
      <input type="number" step="0.01" name="price" value="<?php echo $data['price']; ?>" required class="border p-2 rounded col-span-2">
      <?php if ($data['image']) { ?>
        <div class="col-span-2">
          <p class="mb-2">Gambar Lama:</p>
          <img src="../uploads/<?php echo $data['image']; ?>" class="h-32 rounded mb-2">
        </div>
      <?php } ?>
      <input type="file" name="image" accept="image/*" class="border p-2 rounded col-span-2">
      <button name="update" class="col-span-2 bg-green-600 text-white py-2 rounded">Update Paket</button>
    </form>
    <div class="mt-4">
      <a href="packages.php" class="bg-gray-600 text-white px-4 py-2 rounded">Kembali</a>
    </div>
  </div>
</body>
</html>
